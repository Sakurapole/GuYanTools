# Knowledge Milkdown Markdown POC Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a reversible Milkdown/Crepe proof of concept for Markdown knowledge pages without changing page storage, page types, block editing, or canvas editing.

**Architecture:** Keep CodeMirror as the default Markdown editor and add Milkdown as an experimental editor mode inside the existing Markdown page surface. The POC writes only to `markdownDraft`, saves through the existing `createMarkdownSavePayload(markdown)` path, and can be disabled or removed without migrating database rows.

**Tech Stack:** Electron renderer, Vue 3, TypeScript, Pinia, SCSS, CodeMirror 6, Milkdown Crepe, existing knowledge store and asset APIs.

---

## Scope

This plan implements a POC, not a production migration.

In scope:

- Add Milkdown/Crepe dependencies.
- Add an experimental Milkdown editor component for Markdown pages.
- Add a mode switch in the current Markdown editor toolbar.
- Keep `contentMarkdown` as the only source of truth for Markdown pages.
- Add Markdown round-trip and large-document smoke verification.
- Add enough UI styling for the POC to sit inside the existing knowledge page shell.

Out of scope:

- Do not add a new `pageType`.
- Do not change Rust migrations, Rust services, or N-API contracts.
- Do not replace `KnowledgeBlockEditor.vue` or `KnowledgeCanvasEditor.vue`.
- Do not make Milkdown the default for all Markdown pages.
- Do not remove CodeMirror WYSIWYG or virtual preview.
- Do not support every custom Markdown syntax on day one if the test report records the exact gap.

- `desktop/src/windows/main/stores/knowledge_store.ts`

export function evaluateMilkdownPocRoundTrip(source: string, roundTripped: string, fixture: MilkdownPocFixture): MilkdownPocFixtureResult {
  const missing = fixture.mustContainAfterRoundTrip.filter((needle) => !roundTripped.includes(needle));
  return {
    id: fixture.id,
    passed: source.length > 0 && missing.length === 0,
    missing,
  };
}

export function summarizeMilkdownPocResults(results: MilkdownPocFixtureResult[]) {
  const failed = results.filter((result) => !result.passed);
  return {
    passed: failed.length === 0,
    total: results.length,
    failed: failed.map((result) => result.id),
  };
}

```vue
<script setup lang="ts">
import { nextTick, onBeforeUnmount, onMounted, ref, watch } from 'vue';
import { Crepe } from '@milkdown/crepe';
import '@milkdown/crepe/theme/common/style.css';
import '@milkdown/crepe/theme/frame.css';

const props = defineProps<{
  modelValue: string;
  readonly?: boolean;
  uploadAsset?: (file: File, kind: 'image' | 'attachment') => Promise<string | null>;
}>();

const emit = defineEmits<{
  (event: 'update:modelValue', value: string): void;
}>();

const host = ref<HTMLElement | null>(null);
let editor: Crepe | null = null;
let applyingExternalValue = false;
let lastEmittedMarkdown = props.modelValue;

async function createEditor() {
  if (!host.value || editor) return;

  const crepe = new Crepe({
    root: host.value,
    defaultValue: props.modelValue || '',
    features: {
      [Crepe.Feature.Toolbar]: true,
      [Crepe.Feature.Latex]: true,
    },
    featureConfigs: {
      [Crepe.Feature.Placeholder]: {
        text: '开始写 Markdown',
        mode: 'block',
      },
      [Crepe.Feature.ImageBlock]: {
        async onUpload(file: File) {
          const uploaded = await props.uploadAsset?.(file, 'image');
          return uploaded ?? '';
        },
      },
    },
  });

  crepe.on((listener) => {
    listener.markdownUpdated((_ctx, markdown) => {
      if (applyingExternalValue) return;
      lastEmittedMarkdown = markdown;
      emit('update:modelValue', markdown);
    });
  });

  await crepe.create();
  crepe.setReadonly(Boolean(props.readonly));
  editor = crepe;
}

function destroyEditor() {
  editor?.destroy();
  editor = null;
}

async function replaceEditorValue(value: string) {
  if (!editor || value === lastEmittedMarkdown) return;
  applyingExternalValue = true;
  try {
    destroyEditor();
    await nextTick();
    lastEmittedMarkdown = value;
    await createEditor();
  } finally {
    applyingExternalValue = false;
  }
}

watch(
  () => props.modelValue,
  (value) => {
    void replaceEditorValue(value || '');
  },
);

watch(
  () => props.readonly,
  (value) => {
    editor?.setReadonly(Boolean(value));
  },
);

onMounted(() => {
  void createEditor();
});

onBeforeUnmount(() => {
  destroyEditor();
});
</script>

<template>
  <div class="knowledge-milkdown-poc-editor">
    <div ref="host" class="knowledge-milkdown-poc-editor__host" />
  </div>
</template>

<style scoped lang="scss">
.knowledge-milkdown-poc-editor {

}
```

```ts
function ensureCodeMirrorMode() {
  return mode.value !== 'milkdown';
}

function replaceSelection(before: string, after = before, fallback = '') {
  if (!ensureCodeMirrorMode()) return false;
  const view = editorView.value;
  if (!view) return false;
  const selection = view.state.selection.main;
  const selectedText = view.state.doc.sliceString(selection.from, selection.to) || fallback;
  const insert = `${before}${selectedText}${after}`;
  const anchor = selection.from + before.length;
  const head = anchor + selectedText.length;
  view.dispatch({
    changes: { from: selection.from, to: selection.to, insert },
    selection: { anchor, head },
    scrollIntoView: true,
  });
  view.focus();
  return true;
}
```

```ts
async function handleEditorAssetFile(payload: { file: File; kind: 'image' | 'attachment' }) {
```
